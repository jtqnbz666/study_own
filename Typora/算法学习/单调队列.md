deque 是 push_back() 和 pop_back().

思路：要么以纯单调队列来做，队列中一般记录的是下标(不一定)， 或者看看能不能配合双指针(队列中记录具体的值)

经典滑动窗口最大最小值leetcode 239

~~~c++
#include <bits/stdc++.h>
using namespace std;
int main() {
    int n, k; //n个数， 窗口大小为k
    vector<int>arr(n);
    for(int i = 0; i < n; i ++) cin >> arr[i];
    deque<int>q;
    for(int i = 0; i < n; i ++) {
        while(q.size() && arr[q.back()] > arr[i]) arr.pop_back(); // 把这个 > 换成 <, 那么就是一个单调递减序列
        q.push_back(i);
        
        // 检查队首是否出了窗口范围
        if(i - q.front() == k) q.pop_front();// 好好想一想
        if(i + 1 < k) continue;
        if(i + 1 > k) cout << " "; //小细节
        cout << arr[q.front()];
    }
    cout << endl;
    
    return 0;
}
~~~

滑动窗口c++写法，上一种是记录下标，这种记录具体的值，并且+上了双指针

~~~c++
class Solution {
public:
    deque<int>myque;
    void push(int val) {
        while(!myque.empty() && val > myque.back()) {
            myque.pop_back();
        }
        myque.push_back(val);
    }
    void pop(int val) {
        if(!myque.empty() && val == myque.front()) {
            myque.pop_front();
        }
    }
    int getMaxval() {
        return myque.front();
    }
    vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        vector<int>kk;
        for(int i = 0, j = 0; j < nums.size(); j ++) {
            if(j - i == k) {
               kk.push_back(getMaxval());
               pop(nums[i]);
               i++; 
            }
            push(nums[j]);
            if(j == nums.size() - 1) {
                kk.push_back(getMaxval());
            }
        }
        return kk;
    }
};
~~~

剑指59 || 队列的最大值

~~~c++
class MaxQueue {
public:
    // q表示原队列， mq是单调队列
    deque<int>q, mq;
    MaxQueue() {

    }
    
    int max_value() {
        if(mq.size() == 0) return -1;
        return mq.front();
    }
    
    void push_back(int value) {
        q.push_back(value);
        // 这里不能是 >= ， 因为下边的pop_front()是比对的两个front()值， 如果是 >= ,那么就会出错，  除非把下边pop_front()操作变更为比对下标，但这样就增加了复杂度，需要修改代码
        while(mq.size() && value > mq.back()) mq.pop_back();
        mq.push_back(value);
    }
    
    int pop_front() {
        if(q.size() == 0) return -1;
        if(q.front() == mq.front()) mq.pop_front();
        int ret = q.front();
        q.pop_front();
        return ret;
    }
};
~~~
